```typescript
// src/middleware.ts
import { NextResponse } from 'next/server';
import type { NextRequest } from 'next/server';

const DEMO_MODE = process.env.DEMO_MODE === 'true';

const DEMO_PUBLIC_PATH_PREFIXES = [
  '/auth',
  '/upgrade',
] as const;

function isDemoPublicPath(pathname: string): boolean {
  if (pathname === '/') return true;
  for (const p of DEMO_PUBLIC_PATH_PREFIXES) {
    if (pathname === p || pathname.startsWith(`${p}/`)) {
      return true;
    }
  }
  return false;
}

export function middleware(req: NextRequest) {
  const { pathname } = req.nextUrl;

  const sessionToken =
    req.cookies.get('next-auth.session-token') ??
    req.cookies.get('__Secure-next-auth.session-token');
  const hasSession = Boolean(sessionToken);

  if (DEMO_MODE) {
    if (isDemoPublicPath(pathname)) {
      if (pathname === '/auth/sign-in' && hasSession) {
        return NextResponse.redirect(new URL('/collections', req.url));
      }
      return NextResponse.next();
    }
    if (!hasSession) {
      return NextResponse.redirect(new URL('/', req.url));
    }
    return NextResponse.next();
  }
  // ... non-DEMO_MODE branch ...
}
```

Note: `/auth/auto-demo` will be covered by the existing `DEMO_PUBLIC_PATH_PREFIXES` check because `/auth` is already a prefix. The hero path check must run BEFORE `isDemoPublicPath` is called to ensure unauthenticated visitors get the redirect.

**File 2: `src/app/auth/auto-demo/AutoDemoClient.tsx` (client component)**

```typescript
'use client';

import { useEffect, useState } from 'react';
import { signIn } from 'next-auth/react';

interface AutoDemoClientProps {
  callbackUrl: string;
}

export function AutoDemoClient({ callbackUrl }: AutoDemoClientProps) {
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    async function autoSignIn() {
      try {
        const res = await fetch('/api/demo/credentials');
        if (!res.ok) {
          setError('Demo is not configured. Please contact the site administrator.');
          return;
        }
        const { email, password } = await res.json() as { email: string; password: string };
        await signIn('credentials', {
          email,
          password,
          redirect: true,
          callbackUrl,
        });
      } catch {
        setError('Sign-in failed. Please try refreshing the page.');
      }
    }
    void autoSignIn();
  }, []); // Run once on mount — callbackUrl is stable for the component lifetime

  if (error) {
    return (
      <div className="flex min-h-screen items-center justify-center">
        <div className="text-center space-y-2">
          <p className="text-sm text-muted-foreground">{error}</p>
        </div>
      </div>
    );
  }

  return (
    <div className="flex min-h-screen items-center justify-center">
      <div className="text-center space-y-2">
        <p className="text-sm text-muted-foreground">Loading demo...</p>
      </div>
    </div>
  );
}
```

Key points:
- `'use client'` at line 1 — `signIn` from `next-auth/react` is client-only.
- `useEffect` with empty deps `[]` — runs once on mount.
- `redirect: true` with `callbackUrl` — NextAuth handles the redirect internally after session cookie is committed (use `redirect: true`, NOT `redirect: false`, to avoid race conditions where the session cookie hasn't committed before router navigation).
- Error handling: non-200 from `/api/demo/credentials` → friendly message instead of crash.
- Loading state: simple "Loading demo..." text while the async sign-in runs.
- The `eslint-disable` comment for the exhaustive-deps warning is NOT needed — `callbackUrl` is intentionally excluded from deps because it comes from a server component prop and is stable for the component lifetime. If lint complains, add `// eslint-disable-next-line react-hooks/exhaustive-deps` above the empty deps array.

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| public internet → middleware | Unauthenticated requests hit middleware before any page/API handler |
| middleware → /auth/auto-demo | Middleware issues a redirect; the auto-demo page triggers actual credential sign-in |
| /auth/auto-demo → NextAuth credentials | AutoDemoClient fetches plaintext credentials from /api/demo/credentials and passes them to NextAuth |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-35-03-01 | Authentication Bypass | isHeroPath middleware | mitigate | isHeroPath returns false when PLAYGROUND_COLLECTION_ID is unset — no redirect occurs. The path match uses exact equality (not prefix) to avoid broad route captures. |
| T-35-03-02 | Information Disclosure | /api/demo/credentials | accept | This endpoint is already protected by `isDemoMode()` guard and returns 404 outside DEMO_MODE. Credentials are intentionally public in DEMO_MODE — the demo account is a shared read-only org with no sensitive data. |
| T-35-03-03 | Authentication Bypass | /auth/auto-demo outside DEMO_MODE | mitigate | Server component calls `notFound()` when `isDemoMode()` is false — the page is unreachable in production (D-09). The `isDemoMode()` check runs on the server, not client. |
| T-35-03-04 | Spoofing | callbackUrl redirect | mitigate | `callbackUrl` is set by middleware (trusted server code) and passed as a prop through the server component. The NextAuth `signIn` call with `redirect: true` uses NextAuth's own redirect logic which validates the callbackUrl against the configured `NEXTAUTH_URL`. Open redirect is NextAuth's responsibility — no additional validation needed here. |
| T-35-03-05 | Denial of Service | /auth/auto-demo repeated requests | accept | DEMO_MODE is an opt-in deployment configuration. Rate limiting for the demo path is out of scope for this phase. |
</threat_model>
