**Step 1: Install @radix-ui/react-tooltip**

Run: `npx shadcn-ui@latest add tooltip`

This installs `@radix-ui/react-tooltip` and creates `src/components/ui/tooltip.tsx` with exports: `Tooltip`, `TooltipTrigger`, `TooltipContent`, `TooltipProvider`.

If the shadcn CLI fails (version mismatch), manually:
1. `yarn add @radix-ui/react-tooltip`
2. Create `src/components/ui/tooltip.tsx` following the exact same pattern as `tabs.tsx` (forwardRef, cn() import from `@/lib/utils`, className merge):

```typescript
'use client';

import * as React from 'react';
import * as TooltipPrimitive from '@radix-ui/react-tooltip';
import { cn } from '@/lib/utils';

const TooltipProvider = TooltipPrimitive.Provider;
const Tooltip = TooltipPrimitive.Root;
const TooltipTrigger = TooltipPrimitive.Trigger;

const TooltipContent = React.forwardRef<
  React.ElementRef<typeof TooltipPrimitive.Content>,
  React.ComponentPropsWithoutRef<typeof TooltipPrimitive.Content>
>(({ className, sideOffset = 4, ...props }, ref) => (
  <TooltipPrimitive.Content
    ref={ref}
    sideOffset={sideOffset}
    className={cn(
      'z-50 overflow-hidden rounded-md border bg-popover px-3 py-1.5 text-sm text-popover-foreground shadow-md animate-in fade-in-0 zoom-in-95',
      className
    )}
    {...props}
  />
));
TooltipContent.displayName = TooltipPrimitive.Content.displayName;

export { Tooltip, TooltipTrigger, TooltipContent, TooltipProvider };
```

**Create `src/components/tokens/StyleGuidePanel.tsx`:**

Props interface:
```typescript
interface StyleGuidePanelProps {
  tokens: GeneratedToken[];
  allGroups: TokenGroup[];
  groupName?: string;
}
```

- `tokens`: the tokens for the currently selected group (already filtered by the page)
- `allGroups`: the full group tree needed for `tokenService.resolveTokenReference(value, allGroups)`
- `groupName`: optional display name for the group header

Import `tokenService` from `@/services` and create a local `resolveRef` function:
```typescript
const resolveRef = (value: string): string => {
  return tokenService.resolveTokenReference(value, allGroups);
};
```

**Type-dispatch logic:**

1. First, group tokens by type category:
   - `colorTokens`: tokens where `token.type === 'color'`
   - `spacingTokens`: tokens where `token.type === 'dimension'`
   - `typographyTokens`: tokens where type is one of `'fontFamily' | 'fontSize' | 'fontWeight' | 'lineHeight' | 'letterSpacing' | 'typography'`
   - `shadowTokens`: tokens where type is one of `'boxShadow' | 'shadow' | 'textShadow'`
   - `borderRadiusTokens`: tokens where `token.type === 'borderRadius'`
   - `otherTokens`: all remaining tokens

2. Render in sections, each with a type heading. Only render a section if it has tokens:

```tsx
<div className="p-6 flex flex-col gap-8 overflow-y-auto h-full">
  {colorTokens.length > 0 && (
    <section>
      <h3 className="text-xs font-semibold text-gray-400 uppercase tracking-wider mb-3">Colors</h3>
      <ColorPaletteRow tokens={colorTokens} resolveRef={resolveRef} />
    </section>
  )}
  {spacingTokens.length > 0 && (
    <section>
      <h3 className="text-xs font-semibold text-gray-400 uppercase tracking-wider mb-3">Spacing</h3>
      <div className="flex flex-col gap-2">
        {spacingTokens.map(token => (
          <SpacingPreview key={token.id} token={token} resolvedValue={resolveRef(token.value?.toString() ?? '')} />
        ))}
      </div>
    </section>
  )}
  {typographyTokens.length > 0 && (
    <section>
      <h3 className="text-xs font-semibold text-gray-400 uppercase tracking-wider mb-3">Typography</h3>
      <div className="flex flex-col gap-4">
        {typographyTokens.map(token => (
          <TypographySpecimen key={token.id} token={token} resolvedValue={resolveRef(token.value?.toString() ?? '')} />
        ))}
      </div>
    </section>
  )}
  {shadowTokens.length > 0 && (
    <section>
      <h3 className="text-xs font-semibold text-gray-400 uppercase tracking-wider mb-3">Shadows</h3>
      <div className="flex flex-wrap gap-4">
        {shadowTokens.map(token => (
          <ShadowPreview key={token.id} token={token} resolvedValue={resolveRef(token.value?.toString() ?? '')} />
        ))}
      </div>
    </section>
  )}
  {borderRadiusTokens.length > 0 && (
    <section>
      <h3 className="text-xs font-semibold text-gray-400 uppercase tracking-wider mb-3">Border Radius</h3>
      <div className="flex flex-wrap gap-4">
        {borderRadiusTokens.map(token => (
          <BorderRadiusPreview key={token.id} token={token} resolvedValue={resolveRef(token.value?.toString() ?? '')} />
        ))}
      </div>
    </section>
  )}
  {otherTokens.length > 0 && (
    <section>
      <h3 className="text-xs font-semibold text-gray-400 uppercase tracking-wider mb-3">Other</h3>
      <div className="flex flex-col gap-1">
        {otherTokens.map(token => (
          <TokenValueCard key={token.id} token={token} resolvedValue={resolveRef(token.value?.toString() ?? '')} />
        ))}
      </div>
    </section>
  )}
  {tokens.length === 0 && (
    <div className="text-sm text-gray-400 italic">No tokens in this group</div>
  )}
</div>
```
