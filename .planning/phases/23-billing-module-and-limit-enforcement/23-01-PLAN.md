```typescript
// src/lib/billing/check-collection-limit.ts
// Phase 23 LIMIT-01 — Collection count guard.
// Pattern mirrors src/lib/auth/assert-org-ownership.ts: returns null on success, NextResponse on block.

import { NextResponse } from 'next/server';
import dbConnect from '@/lib/mongodb';
import Organization from '@/lib/db/models/Organization';
import TokenCollection from '@/lib/db/models/TokenCollection';
import { LIMITS, type PlanTier } from './tiers';

export async function checkCollectionLimit(
  organizationId: string
): Promise<NextResponse | null> {
  // D-14: SELF_HOSTED bypass — short-circuit BEFORE any DB read
  if (process.env.SELF_HOSTED === 'true') return null;

  if (!organizationId) {
    return NextResponse.json({ error: 'Not Found' }, { status: 404 });
  }

  await dbConnect();

  const org = await Organization.findById(organizationId)
    .select('planTier')
    .lean() as { planTier?: PlanTier } | null;

  if (!org) {
    return NextResponse.json({ error: 'Not Found' }, { status: 404 });
  }

  const tier: PlanTier = org.planTier ?? 'free';
  const max = LIMITS[tier].maxCollections;

  // Pitfall 7 — Infinity is not a valid MongoDB query value; short-circuit in JS.
  if (max === Infinity) return null;

  // D-13: read current count; check in JS (not via Mongo $lt — see Pitfall 7).
  // For POST /api/collections this is low traffic so the read-then-create race window
  // is acceptable. For high-frequency counters (exports) Plan 02 uses findOneAndUpdate.
  const current = await TokenCollection.countDocuments({ organizationId });

  if (current >= max) {
    return NextResponse.json(
      { code: 'LIMIT_EXCEEDED', resource: 'collections', current, max, tier },
      { status: 402 }
    );
  }

  return null;
}
```

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-23-01-01 | Tampering | Client-sent `organizationId` | mitigate | `organizationId` read from `session.user.organizationId` only (never body/query); `assertOrgOwnership` already enforces this pattern on collection routes; POST handler uses `getServerSession(authOptions)` for org id — acceptance criteria forbid reading from request body |
| T-23-01-02 | Elevation of Privilege | Stale JWT planTier | mitigate | `checkCollectionLimit` queries Organization doc fresh on every call (`Organization.findById(...).select('planTier')`); JWT is never read for plan tier (Pitfall 3); acceptance criteria grep the guard implementation for `Organization.findById` |
| T-23-01-03 | Denial of Service | Unbounded collection creation | mitigate | Atomic count check + 402 response before `repo.create`; LIMIT-01 caps free tier at 1; team tier returns Infinity short-circuit to avoid unnecessary counts |
| T-23-01-04 | Information Disclosure | Cross-tenant org probing via planTier response | mitigate | 404 returned when org not found (no leak); 402 payload contains only the caller's own tier/usage (already authorised by `requireRole`); empty `organizationId` returns 404 not 500 |
| T-23-01-05 | Repudiation | Developer-mode bypass in production | mitigate | `SELF_HOSTED=false` explicit default in `.env.local.example`; env var literal string must equal `'true'` (case-sensitive) to bypass; any other value = limits active |
